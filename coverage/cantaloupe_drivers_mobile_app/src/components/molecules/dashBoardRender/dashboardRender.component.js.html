/**
 * Dashboard flatlist render component as an Molecule...
 *
 */
import React from 'react';
import { View } from 'react-native';
import { heightPercentageToDP as hp, widthPercentageToDP as wp } from 'react-native-responsive-screen';
import * as colors from '../../../constants/uiConstants/colorConstants';
import { IconComponent, TextComponent, TouchableItem } from '../../atoms';
import MaterialCommunityIcons from 'react-native-vector-icons/MaterialCommunityIcons';
import styles from './dashboardRenderer-styles';
import acceessbilityKey from '../../../constants/accessibilityConstants';
 
const DashBoardRender = (props) => {
    const { menuItem, selectItems } = props;
    const iconWidth = menuItem?.menuName === 'Pick List' ? hp(6.4) : menuItem?.menuName === 'Dex' ? hp(8.5) : hp(7);
    const statusEnum = { "1": 'tick', "2": 'exclamation' };
 
    const exclamation = menuItem?.status === statusEnum[2] && "true"
    const tickStatus = menuItem?.status === statusEnum[1] && "true"
    const mdisble = menuItem?.Permission && "true"
    let testValues = "exclamation: " + exclamation + ";" + "tick: " + tickStatus + ";" + "menu: " + mdisble
    return (
        <View style={styles.maincontainer}>
            <TouchableItem
                onPress={() => menuItem?.Permission ? selectItems(menuItem) : null}
                activeOpacity={menuItem?.Permission ? 0.5 : 1}
                testKey={menuItem?.menuName}
                testVal={testValues}
                style={styles.menuBox}
            >
                <View
                    style={[menuItem?.status == statusEnum[1] && styles.circleBadge, menuItem?.status == statusEnum[2] && styles.circleBadge, styles.circleBox, { borderColor: menuItem?.Permission ? colors.BLUE : colors.BLUELIGHTER }]}>
                    {menuItem?.status === statusEnum[2] &&
                        <View style={styles.badge} >
                            <MaterialCommunityIcons accesible={true} name="exclamation-thick" size={wp(5)} color={colors?.WHITE} />
                        </View>
                    }
                    {menuItem?.status === statusEnum[1] &&
                        <View a style={styles.badgeGrey}>
                            <MaterialCommunityIcons name="check" size={wp(5)} color={colors?.WHITE} />
                        </View>
                    }
                    {menuItem?.Permission ? (
                        <IconComponent width={iconWidth} height={iconWidth} icon={menuItem?.iconShow} />
                    ) : (
                        <IconComponent width={hp(7)} height={hp(7)} icon={menuItem?.iconBlur} />
                    )}
                </View>
                <TextComponent testVal={menuItem?.menuName} testResID={testValues} style={[styles.menuText, { color: menuItem?.Permission ? colors.BLUE : colors.BLUELIGHTER }]}>{menuItem?.menuName}</TextComponent>
            </TouchableItem>
        </View >
    );
};
 
export default DashBoardRender;
