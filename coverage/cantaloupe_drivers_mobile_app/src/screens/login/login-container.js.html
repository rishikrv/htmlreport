import React, { useState, useEffect } from 'react';
import { Keyboard } from 'react-native';
import { useIsFocused, useNavigation } from '@react-navigation/native';
import LoginScreen from './login-screen';
import NavigationConstants from '../../navigator/navconstants/navigation-constants';
import { netInfo, emailValidation, flashMessage } from '../../utils';
import stringConstants from '../../constants/uiConstants/stringConstants';
import { realmStoreAction, realmFetchAction } from '../../realm/realmAction';
import { callSetApiDataOperator, callSetApiDataRoute, callSetApiDataMachine } from '../../utils/apiOperation';
import base64 from 'react-native-base64';
import { StoreTokenSuccess } from '../../redux/actions/loginAction';
import { StoreSelectedOperatorSuccess } from '../../redux/actions/selectedValueAction';
import { useDispatch } from 'react-redux';
import mixpanel from '../../components/atoms/mixPanel/mixpanel.component';
const { INTERNET_WARN, OKAY, } = stringConstants;
const Login = props => {
  const navigation = useNavigation();
  const isFocused = useIsFocused();
 
  const dispatch = useDispatch();
 
  const [emailId, setEmailId] = useState('');
  const [password, setPassword] = useState('');
  const [emailIdFocus, setEmailIdFocus] = useState(false);
  const [isLocalError, setIsLocalError] = useState(null);
  const [isServerError, setIsServerError] = useState(null);
  const [loader, setLoader] = useState(false);
  const [modal, setModal] = useState(false);
  const [modalName, setModalName] = useState(null);
  const [isNetActive, setIsNetActive] = useState(true);
 
  useEffect(() => {
    netInfo(setIsNetActive);
  }, [isNetActive]);
 
  useEffect(() => {
    fetchRealm();
  }, [isFocused]);
 
  const fetchRealm = async () => {
    let userData = await realmFetchAction.userName()
    let savedEmail = userData[0]?.email;
    if (savedEmail?.length > 0) {
      setEmailId(savedEmail);
    }
  };
 
  // onsubit will be called when the user clicks on submit button
 
  const onSubmit = async values => {
    Keyboard.dismiss()
    if (isNetActive) {
      const encodedData = `${base64.encode(`${emailId}:${password}`)}`;
      Keyboard.dismiss();
      dispatch(StoreTokenSuccess(encodedData));
      var mailformat = emailValidation();
      const userName = emailId?.length ? emailId?.trim() : emailId;
      const userPassword = password?.length ? password?.trim() : password;
      if ((userName.length === 0) && userPassword.length === 0) {
        setIsLocalError(stringConstants.EMPTY_CREDENTIALS);
      }
      else if ((userName.length && !userName.match(mailformat)) && userPassword.length === 0) {
        setIsLocalError(stringConstants.INVALID_CREDENTIALS)
      }
      else if (!userName.match(mailformat) || userName.length === 0) {
        setIsLocalError(stringConstants.INVALID_MAILID);
      } else if (userPassword.length === 0) {
        setIsLocalError(stringConstants.INVALID_PASSWORD);
      } else {
        setLoader(true);
        let data = [{ email: emailId }];
        realmStoreAction.userName(data, res => { });
        if (!loader) apiCall(emailId, password);
      }
    }
    else {
      flashMessage(stringConstants?.INTERNET_WARN, "", "danger", false)
    }
  };
 
  // to call all api and navigate
 
  const apiCall = () => {
    callSetApiDataOperator((res) => {
      if (res?.status === 200) {
		  const userName = `${emailId}`
		  mixpanel.identify(userName);
		  mixpanel.track(userName)
		  mixpanel.registerSuperProperties({ "Account type": "Free", "User Type": userName });
        setLoader(false);
        if (res?.data?.Operators?.length === 1) {
          const selectedOp = res?.data?.Operators[0].Code;
          dispatch(StoreSelectedOperatorSuccess(selectedOp));
          callSetApiDataRoute(route => {
            if (route?.data?.length === 1) {
              if (route?.data?.Schedules?.length > 1) {
                handleNavigation();
                setLoader(false);
              }
              else {
                const schduledId = { "ScheduleId": route?.data?.Schedules[0]?.Id }
                callSetApiDataMachine(schduledId, (machine) => {
                  if (machine?.status !== undefined) {
                    setLoader(false);
                    handleNavigation();
                  }
                });
              }
            }
            else {
              handleNavigation();
              setLoader(false);
            }
          });
        } else {
          handleNavigation();
          setLoader(false);
        }
      }
      else {
        setLoader(false);
        setIsServerError(stringConstants.ERROR);
      }
    });
  };
 
  const handleNavigation = async () => {
    let operatorData = await realmFetchAction.operator()
    let routeData = await realmFetchAction.route()
 
    if (operatorData[0]?.Operators?.length > 1) {
      onNavigate("Operator");
    }
    // if multiple route navigate to route else multiple date for route go date else asset
    else if (routeData?.length > 1) {
      onNavigate("Route", routeData);
    }
    else {
      // / if multiple date navigate to date else asset
      if (routeData[0]?.Schedules?.length > 1) {
        onNavigate("Date", routeData);
      }
      else {
        onNavigate("Asset", routeData);
      }
    }
    Keyboard.dismiss();
  }
  const onNavigate = (navigateTo, routeData) => {
    switch (navigateTo) {
      case "Operator":
        navigation.replace(NavigationConstants.MAIN_STACK, {
          screen: NavigationConstants.OPERATORSLECTOR_SCREEN,
        });
        break;
      case "Route":
        navigation.replace(NavigationConstants.MAIN_STACK, {
          screen: NavigationConstants.ROUTESELECTOR_SCREEN,
        });
        break;
      case "Date":
        navigation.replace(NavigationConstants.MAIN_STACK, {
          screen: NavigationConstants.SELECT_SCHEDULE,
          selectedRoute: routeData[0]?.Schedules,
          routeName: routeData[0]?.Name,
        });
        break;
      case "Asset":
        navigation.replace(NavigationConstants.MAIN_STACK, {
          screen: NavigationConstants.SELECT_MACHINE,
          selectedRoute: routeData[0]?.Schedules[0],
          routeStatus: true,
          routeName: routeData[0]?.Name,
          selectedDate: routeData[0]?.Schedules[0],
        });
        break;
      default:
        alert("Something went wrong.")
    }
  }
 
 
  // submitText will be called when the user clicks on keyboard done
 
  const submitText = () => {
    Keyboard.dismiss();
  };
 
  const handleEmailChange = value => {
    setEmailId(value);
    setIsServerError(null);
    setIsLocalError(null);
  };
 
  const handlePasswordChange = value => {
    setPassword(value);
    setIsServerError(null);
    setIsLocalError(null);
  };
  const handleBackdrop = () => {
    setModal(false);
    setModalName(null);
  };
  const getModalChild = (name) => {
    switch (name) {
      case 'noInternet':
        return (
          {
            handleYes: handleBackdrop,
            textFirst: INTERNET_WARN,
            buttonTextFirst: OKAY,
          }
        );
      default:
        return (
          [{}]
        );
    }
  };
  const modalChild = getModalChild(modalName);
  return (
    <>
      <LoginScreen
        onSubmit={onSubmit}
        emailId={emailId}
        password={password}
        emailIdFocus={emailIdFocus}
        setEmailIdFocus={setEmailIdFocus}
        maxlength={10}
        handleUnhandledTouches={submitText}
        isLocalError={isLocalError}
        isServerError={isServerError}
        handleEmailChange={handleEmailChange}
        handlePasswordChange={handlePasswordChange}
        loader={loader}
        modal={modal}
        modalProps={modalChild}
        handleBackdrop={handleBackdrop}
        {...props}
      />
    </>
  );
};
 
export default Login;
